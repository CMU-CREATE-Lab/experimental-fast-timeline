/**
 * Polyfill for emulating Math.log2 function
 * from http://mzl.la/1Ne7YQn
 * @param  {number} x
 * @return {number}
 */

Math.log2 = Math.log2 ||
            function(x) {
                return Math.log(x) / Math.LN2;
            };
